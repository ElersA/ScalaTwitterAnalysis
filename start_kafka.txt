% Export environment variables
export KAFKA_HOME="/path/to/the/kafka/folder"
export PATH=$KAFKA_HOME/bin:$PATH

% Start zookeeper server
$KAFKA_HOME/bin/zookeeper-server-start.sh $KAFKA_HOME/config/zookeeper.properties

% Start Kafka
$KAFKA_HOME/bin/kafka-server-start.sh $KAFKA_HOME/config/server.properties

% Create a Kafka topic called "twitter"
$KAFKA_HOME/bin/kafka-topics.sh --create --zookeeper localhost:2181 --replication-factor 1 --partitions 1 --topic twitter

% Add this to config/server.properties to be able to delete topics easily (facilitates testing)
delete.topic.enable=true

% Delete topics with the following command
$KAFKA_HOME/bin/kafka-topics.sh --zookeeper localhost:2181 --delete --topic twitter
